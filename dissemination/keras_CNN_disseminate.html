Binary classification with keras starting from Cerenkov calibrated data¶

It'll be a session of documented code to reach the goal. I'll work under a conda environment, with keras + tensofrlow-gpu installed. I could work with 16-bit images then exploiting the handsome keras functions ImageDataGenerator and its flow_from_directory, which handles different populations just by arranging training files in proper directories; indeed that was the first trial, and it worked very fine. But since (at the time this work was performed) this feature was not available for regression problems (and the full reconstruction procedure for telescope-tracked particles involve regressions on energy and arrival direction), I went back and handled all by custom functions, following this example.

Since I wrote those function on my own, I decided not to work with images, but with their digital translation: matrixes. In other words, there's no need to produce an image from raw data just to decode it again with any suitable function to have 2D matrixes: I could produce those matrixes at first, and so I did.

I won't describe the procedure to produce calibrated matrixes from raw data, which is not exactly machine learning indeed.

So, let's start importing some useful modules (did I mention I love python?).

def conv_model(img_width, img_height, channels=3, channel_pos='channels_last', batch_norm=False):

    K.set_image_data_format(channel_pos)

    if K.image_data_format() == 'channels_first':
        input_shape = (channels, img_width, img_height)
        norm_axis = 1
    else:
        input_shape = (img_width, img_height, channels)
        norm_axis = -1

    model = Sequential()
    model.add(Conv2D(32, (3, 3), input_shape=input_shape))
    if batch_norm:
        model.add(BatchNormalization(axis=norm_axis))
    model.add(Activation('relu'))
    model.add(MaxPooling2D(pool_size=(2, 2)))
    model.add(Conv2D(32, (3, 3)))
    if batch_norm:
        model.add(BatchNormalization(axis=norm_axis))
    model.add(Activation('relu'))
    model.add(MaxPooling2D(pool_size=(2, 2)))
    model.add(Conv2D(64, (3, 3)))
    if batch_norm:
        model.add(BatchNormalization(axis=norm_axis))
    model.add(Activation('relu'))
    model.add(MaxPooling2D(pool_size=(2, 2)))
    model.add(Flatten())
    model.add(Dense(64))
    if batch_norm:
        model.add(BatchNormalization(axis=norm_axis))
    model.add(Activation('relu'))
    if batch_norm:
        model.add(Dropout(0.2))
    else:
        model.add(Dropout(0.5))
    # if last_act == 'sigmoid':
    #     model.add(Dense(1))
    # else:
    #     model.add(Dense(2))
    #
    # model.add(Activation(last_act))

    return model

You can see the last layer is commented out: I preferred to give the user the freedom to choose it in the training function interface.

Managing input files to be ready for training¶

As already mentioned, I needed to mimick the flow_from_directory behaviour by hand, so I create two dictionaries: the id_dict has the training and test filename lists, while the labels_dict maps a binary label to each filename.

Here it is:

def create_partition_labels_dict(train_path, test_path):
    """Generates events ID list and their labels"""
    # read ids
    gamma_train_ids = glob.glob(train_path + '/gammas/*.npz')
    proton_train_ids = glob.glob(train_path + '/protons/*.npz')
    gamma_test_ids = glob.glob(test_path + '/gammas/*.npz')
    proton_test_ids = glob.glob(test_path + '/protons/*.npz')

    train_ids = gamma_train_ids + proton_train_ids
    test_ids = gamma_test_ids + proton_test_ids
    # build ids dictionary
    id_dict = {'train': train_ids, 'test': test_ids}

    # make labels
    gamma_train_labels = [1 for _ in range(len(gamma_train_ids))]
    proton_train_labels = [0 for _ in range(len(proton_train_ids))]
    gamma_test_labels = [1 for _ in range(len(gamma_test_ids))]
    proton_test_labels = [0 for _ in range(len(proton_test_ids))]

    train_labels = gamma_train_labels + proton_train_labels
    test_labels = gamma_test_labels + proton_test_labels

    # build labels dictionary
    labels1 = dict(zip(train_ids, train_labels))
    labels2 = dict(zip(test_ids, test_labels))
    # merge the two dicts
    labels_dict = {**labels1, **labels2}

    return id_dict, labels_dict

def train_model(train_path, test_path, last_act_layer='sigmoid'):
    # training_path = '/astri03/data2/visconti/DL1/DC1/camera_data/train/'
    # testing_path = '/astri03/data2/visconti/DL1/DC1/camera_data/test/'

    def rmse(y_true, y_pred):
        return K.sqrt(K.mean(K.square(y_pred - y_true), axis=-1))

    params = {'width': 56,
              'height': 56,
              'batch_size': 20,
              'shuffle': True}

    epochs = 50

    logger.info("Generating IDs and labels dictionaries...")
    # pdb.set_trace()
    partition, labels = create_partition_labels_dict(train_path, test_path)
    logger.info("Done.")

    training_generator = DataGenerator(**params).generate(labels, partition['train'])
    validation_generator = DataGenerator(**params).generate(labels, partition['test'])

    model = conv_model(params['width'], params['height'], channels=1, batch_norm=True)
    if last_act_layer == 'sigmoid':
        model.add(Dense(1))
        model.add(BatchNormalization(axis=-1))
        model.add(Activation(last_act_layer))
        model.compile(loss='binary_crossentropy', optimizer='adam', metrics=['accuracy', rmse])
    else:
        model.add(Dense(2))
        model.add(BatchNormalization(axis=-1))
        model.add(Activation(last_act_layer))
        model.compile(loss='sparse_categorical_crossentropy', optimizer='adam', metrics=['accuracy', rmse])

    filepath = last_act_layer + "_weights_matrix.best.hdf5"
    checkpoint = ModelCheckpoint(filepath, monitor='val_acc', verbose=1, save_best_only=True, mode='max')
    earlystopping = EarlyStopping(monitor='val_acc', patience=3, min_delta=0.001)
    callbacks_list = [checkpoint, earlystopping]

    logger.info("Train now!")
    start = time.process_time()
    summary = model.fit_generator(training_generator, steps_per_epoch=len(partition['train']) // params['batch_size'],
                                  epochs=epochs, validation_data=validation_generator,
                                  validation_steps=len(partition['test']) // params['batch_size'],
                                  callbacks=callbacks_list)

    stop = time.process_time()
    logger.info("Time to fit the model: {}".format(stop - start))

    return summary

Worth to mention:

  • I build two generators, one for training and one for test files;
  • I have 'sigmoid' as default last_act_layer since it is the obvious choice for binary classification;
  • I use 'adam' optimizer because it's probably the most popular: experiments needs to be done with other optimizers (but I probably would change network first);
  • I use checkpoints on validation accuracy, and earlystopping on the same parameter with a tolerance of $0.001$: if the validation accuracy does not improve by a factor of at least $0.001$ in three epochs, the execution stops.
  • Using DataGenerator (although a custom one) allows to use the fit_generator method of keras.models which is more than beautiful.

def prediction_labels_dict(path):
    """Generates events ID list and their labels"""
    # read ids
    gamma_ids = glob.glob(path + '/gammas/*.npz')
    proton_ids = glob.glob(path + '/protons/*.npz')

    ids = gamma_ids + proton_ids
    # build ids dictionary
    id_dict = {'pred': ids}

    # make labels
    gamma_labels = [1 for _ in range(len(gamma_ids))]
    proton_labels = [0 for _ in range(len(proton_ids))]

    pred_labels = gamma_labels + proton_labels

    # build labels dictionary
    labels_dict = dict(zip(ids, pred_labels))

    return id_dict, labels_dict

def keras_dnn(train_path, validation_path, img_width, img_height, img_channels, last_act_layer='sigmoid', batch_norm=False):


    from keras.preprocessing.image import ImageDataGenerator
    from keras.callbacks import ModelCheckpoint, EarlyStopping
    from keras.layers import Activation, Dense, BatchNormalization
    from keras import backend as K

    # reproducibility
    np.random.seed(42)
    from tensorflow import set_random_seed
    set_random_seed(4242)

    nb_train_samples = 70000 * 2
    nb_validation_samples = 30000 * 2
    epochs = 30
    batch_size = 16

    model = conv_model(img_width, img_height, batch_norm=batch_norm, channels=img_channels)

    def rmse(y_true, y_pred):
        return K.sqrt(K.mean(K.square(y_pred - y_true), axis=-1))

    if K.image_data_format() == 'channels_first':
        norm_axis = 1
    else:
        norm_axis = -1

    # model.load_weights('first_try.h5')

    if last_act_layer == 'sigmoid':
        model.add(Dense(1))
        if batch_norm:
            model.add(BatchNormalization(axis=norm_axis))
        model.add(Activation(last_act_layer))
        model.compile(loss='binary_crossentropy', optimizer='adam', metrics=['accuracy', rmse])
    else:
        model.add(Dense(2))
        if batch_norm:
            model.add(BatchNormalization(axis=norm_axis))
        model.add(Activation(last_act_layer))
        model.compile(loss='sparse_categorical_crossentropy', optimizer='adam', metrics=['accuracy', rmse])

    train_datagen = ImageDataGenerator(rescale=1. / 255)

    test_datagen = ImageDataGenerator(rescale=1. / 255)

    train_generator = train_datagen.flow_from_directory(train_path,
                                                        target_size=(img_width, img_height),
                                                        batch_size=batch_size,
                                                        class_mode='binary',
                                                        color_mode='grayscale',
                                                        shuffle=True)

    validation_generator = test_datagen.flow_from_directory(validation_path,
                                                            target_size=(img_width, img_height),
                                                            batch_size=batch_size,
                                                            class_mode='binary',
                                                            color_mode='grayscale',
                                                            shuffle=True)

    filepath = last_act_layer + "_images_weights.best.hdf5"
    checkpoint = ModelCheckpoint(filepath, monitor='val_acc', verbose=1, save_best_only=True, mode='max')
    earlystopping = EarlyStopping(monitor='val_acc', patience=5, min_delta=0.001)
    callbacks_list = [checkpoint, earlystopping]

    start = time.process_time()
    summary = model.fit_generator(train_generator, steps_per_epoch=nb_train_samples // batch_size,
                                  epochs=epochs, validation_data=validation_generator,
                                  validation_steps=nb_validation_samples // batch_size,
                                  callbacks=callbacks_list)

    stop = time.process_time()
    logger.info("Time to fit the model: {}".format(stop - start))

    # model.save_weights('first_try.h5')

    return summary, train_generator.class_indices

def keras_predict_gen(predict_path, img_width, img_height, img_channels, last_act_layer='sigmoid', batch_norm=False):
